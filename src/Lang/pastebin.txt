/* Example Program to print the first n numbers of the Fibonacci Sequence */

// Main Function (execution starts here)
void main()
{
    let int n = (int)input("Enter n: ");

    let int[] sequence = fib(n);

    print("The first {n} numbers of the Fibonacci Sequence is : { (string)sequence }"); 
    // Example: The first 8 numbers of the Fibonacci Sequence is : [0, 1, 1, 2, 3, 5, 8, 13]
}

// Fib Function (Generated Sequence)
int[] fib(int n)
{
    let int[n] sequence; // Arrays can also be declared with a value (e.g. int[4] x = {1, 2, 3, 4} )

    for (let int i = 0; i < n; i++)
    {
        if (i == 0)
        {
            sequence += 0; // Adding an element to an array (type conflicts will result in error)
            continue;
        } 

        if (i == 1)
        {
            sequence += 1;
        }
        else
        {
            sequence += sequence[-1] + sequence[-2]
            
            /* 
            NB: Compiler / VM will have to check that pointer arithmetic 
            doesn't overreach the bounds of the array (by tracking)
            array length and injecting code into the final binary, in
            areas where errors could occur.
            */
        }
    }

    return sequence;
}
